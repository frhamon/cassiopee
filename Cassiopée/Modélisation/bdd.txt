CREATE DATABASE cassiopee CHARACTER SET 'utf8';



CREATE TABLE IF NOT EXISTS ICSCERT (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
date DATE, 
patch INT)  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS Patch (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
patch CHAR(255))  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS CVE (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
date DATE, 
advisoryDescription TEXT, 
cvss INT, 
cwe INT)  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS CWE (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
cweLink,
cweAbstraction, 
cweStructure, 
cweSFP2Cluster INT)  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS SFP2 (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
name CHAR(255), 
sfp1 INT)  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS SFP1 (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
name CHAR(255))  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS CVSS (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
AV VARCHAR(1), 
AC VARCHAR(1), 
PR VARCHAR(1), 
UI VARCHAR(1), 
S VARCHAR(1), 
C VARCHAR(1), 
I VARCHAR(1), 
A VARCHAR(1))  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS Product (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
name CHAR(255), 
sector INT, 
countriesDeployed INT, 
deviceYN TINYINT, 
deviceType INT, 
deviceComment TEXT, 
vendor INT) ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS Vendor (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
location CHAR(255))  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS DeviceType (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
type CHAR(255))  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS Sector (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
name CHAR(255))  ENGINE = INNODB;



CREATE TABLE IF NOT EXISTS Countries (
id SMALLINT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY, 
code VARCHAR(2) NOT NULL default '', 
name CHAR(255) NOT NULL default '')  ENGINE = INNODB;



ALTER TABLE CVE ADD CONSTRAINT fk_cve_cvss FOREIGN KEY (cvss) REFERENCES CVSS(id);



ALTER TABLE CVE ADD CONSTRAINT fk_cve_cwe FOREIGN KEY (cwe) REFERENCES CWE(id);



ALTER TABLE SFP2 ADD CONSTRAINT fk_sfp2_sfp1 FOREIGN KEY (sfp1) REFERENCES SFP1(id);



ALTER TABLE CWE ADD CONSTRAINT fk_cwe_cwesfp2cluster FOREIGN KEY (cweSFP2CLuster) REFERENCES SFP2(id);



ALTER TABLE ICSCERT ADD CONSTRAINT fk_icscert_patch FOREIGN KEY (patch) REFERENCES Patch(id);



ALTER TABLE Product ADD CONSTRAINT fk_product_sector FOREIGN KEY (sector) REFERENCES Sector(id);



ALTER TABLE Product ADD CONSTRAINT fk_product_deviceType FOREIGN KEY (deviceType) REFERENCES DeviceType(id);



ALTER TABLE Product ADD CONSTRAINT fk_product_vendor FOREIGN KEY (vendor) REFERENCES Vendor(id);



CREATE TABLE ICSCERT_CVE
(
	id_icscert_cve INT NOT NULL IDENTITY CONSTRAINT PK_icscert_cve PRIMARY KEY,
	id_icscert INT NOT NULL CONSTRAINT FK_icscert_cve__id_icscert FOREIGN KEY (id_icscert) REFERENCES ICSCERT(id),
	id_cve INT NOT NULL CONSTRAINT FK_icscert_cve__id_cve FOREIGN KEY (id_cve) REFERENCES CVE(id),
	CONSTRAINT UQ_icscert_cve__id_icscert__id_cve UNIQUE (id_icscert, id_cve)
);



CREATE TABLE ICSCERT_Product
(
	id_icscert_product INT NOT NULL IDENTITY CONSTRAINT PK_icscert_product PRIMARY KEY,
	id_icscert INT NOT NULL CONSTRAINT FK_icscert_product__id_icscert FOREIGN KEY (id_icscert) REFERENCES ICSCERT(id),
	id_product INT NOT NULL CONSTRAINT FK_icscert_product__id_product FOREIGN KEY (id_product) REFERENCES Product(id),
	CONSTRAINT UQ_icscert_product__id_icscert__id_product UNIQUE (id_icscert, id_product)
);



CREATE TABLE Product_Countries
(
	id_product_countries INT NOT NULL IDENTITY CONSTRAINT PK_product_countries PRIMARY KEY,
	id_product INT NOT NULL CONSTRAINT FK_product_countries__id_product FOREIGN KEY (id_product) REFERENCES Product(id),
	id_countries INT NOT NULL CONSTRAINT FK_product_countries__id_countries FOREIGN KEY (id_countries) REFERENCES Countries(id),
	CONSTRAINT UQ_product_countries__id_product__id_countries UNIQUE (id_product, id_countries)
);
